import { Meta, ColorPalette, ColorItem } from "@storybook/addon-docs/blocks";

export const makePalletteVariables = (baseName) => (
  [50, 100, 200, 300, 400, 500, 600, 700, 800, 900, 950].map(
    (shade) => `--color-${baseName}-${shade}`
  )
)

export const mapCssVariableToPaletteItem = (cssVariable) => {
  let token = cssVariable.replace(/^--(?:color|background-color)-/, "")
  
  const title = 
    token.split('-').map(part => 
      part.charAt(0).toUpperCase() + part.slice(1)
    ).join(' ')
    
  const colors = [`var(${cssVariable})`]
  
  return <ColorItem title={title} colors={colors} />
}

<Meta title="Colors" />

<div
  style={Object.fromEntries(makePalletteVariables('mud').map((cssVariable, index) => [
      `--${index}`,
      `var(${cssVariable})`
    ]))}
/>

# Colors

Lumo theme is built on top of Tailwind CSS.

Since theme is extended, 
[all Tailwind default colors](https://tailwindcss.com/docs/colors) are 
available.

In addition to that, colors were tokenized to provide them meaning and 
semantics.

## Color pallettes

Lumo theme defines three color pallettes.

### Mud

<ColorPalette>
  {makePalletteVariables('mud').map(mapCssVariableToPaletteItem)}
</ColorPalette>

### Sand

<ColorPalette>
  {makePalletteVariables('sand').map(mapCssVariableToPaletteItem)}
</ColorPalette>

### Vermilion

<ColorPalette>
  {makePalletteVariables('vermilion').map(mapCssVariableToPaletteItem)}
</ColorPalette>

## Content colors

These colors are applicable for color utilities besides background.

<ColorPalette>
  {[

    '--background-color-elevated',
    '--color-accent',
  ].map(mapCssVariableToPaletteItem)}
</ColorPalette>
